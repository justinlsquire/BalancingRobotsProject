\section{Verbinden des Roboters}
\begin{frame}{Verbinden des Roboters: Hardware}
	\begin{itemize}
		\item Setzen Sie sechs AA Batterien auf der Rückseite des Roboters ein.
		\item Schlie\ss{}en Sie den Roboter mit Hilfe des USB-Kabels an Ihren Rechner an. 
		\item Setzen Sie  ``Driver Enable''an der oberen rechten Ecke auf ``off''.
		\end{itemize}
\end{frame}

\begin{frame}{Verbinden des Roboters: Öffnen der Software}
	Gehen Sie zum GitHub-Verzeichnis \texttt{BalancingRobotsProject $\mapsto$ Arduino $\mapsto$ Minseg\_test\_M2V5}, öffnen Sie die Datei ``Minseg\_test\_M2V5.ino'' in Arduino IDE.
	\begin{center}
		\includegraphics[width = 0.9\textwidth]{github-repository-explorer-minseg-test}
	\end{center}
\end{frame}


\begin{frame}{Verbinden des Roboters: Wählen des Boards}
	Wählen Sie im Menu von Ardioni IDE die Option \texttt{Tools $\mapsto$ Board} und wählen  ``Arduino/Genuino Mega or Mega 2560'' oder ``Arduino Mega or Mega 2560'' aus:
	\begin{center}
		\includegraphics[width = 0.5\textwidth]{arduino-board-manager}
	\end{center}
\end{frame}


\begin{frame}{Verbinden des Roboters: Wählen des Ports}
	Wählen Sie im Menu von Ardioni IDE die Option \texttt{Tools $\mapsto$ Port} und wählen Sie den COM port aus, an dem der MinSeg angeschlossen ist.
	\begin{center}
		\includegraphics[width = 0.5\textwidth]{arduino-board-tool}
	\end{center}
	(Wenn Sie den MinSeg vom Rechner trennen und wieder anschlie\ss{}en, sollte ein Eintrag in der Liste neu hinzukommen. Dies ist der Port, den Sie auswählen sollten.)
\end{frame}
